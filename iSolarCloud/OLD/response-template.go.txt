package domain

import (
	"GoSungro/Once"
	"GoSungro/pbx/web"
	"fmt"
)


func (t *CountResponse) String() string {
	var str string
	for range Once.Only {
		str = fmt.Sprintf("%d", t.Total)
	}

	return str
}
func (t *CountResponse) JsonString() string {
	var str string
	for range Once.Only {
		j, err := web.PrintAsJson(t)
		if err != nil {
			break
		}
		str = fmt.Sprintf(j)
	}

	return str
}
